//print fiboncciseries using variadic template





#include <iostream>

using namespace std;

template<typename arg>
int fib(arg val)
{
    return val;
}

template<typename arg, typename...args>
int fib(arg val, args... values )
{
    return(val * fib(values...));
}


int main()
{
    cout<<"fib data is: "<<fib(1,2,3,4,5);
//  cout<<"Hello World";
    return 0;
}
