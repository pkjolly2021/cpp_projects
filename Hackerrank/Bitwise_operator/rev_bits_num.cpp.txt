using namespace std;


void rev_bits(unsigned int dat)
{
    unsigned int ans = 0;
    int cnt = 1;
    while(cnt <= 32)
    {
        ans <<= 1;
        ans |= (dat&1);
        dat >>= 1;
      
        cnt++;
    }
cout<<"rev data is: "<<ans<<std::endl;
}
