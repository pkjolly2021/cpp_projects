//TO GEGT THE PREVIOUS POWER OF 2.
// FOR LONG INT 4 BYTE, SETTING ONLY LEFT MOST BIT IN CURRENT NUMBER.

//EX: 9 1001
//      1000 - is the power of 2 previous to 9.


int find_prev_numb_power_2(long n)
{
   long val;
   long prev_num = 0x80000000;
   do
   {
      if (val = (prev_num  & n))
       {
           break;
       }
       prev_num >>= 1;
   }while(prev_num);
    val = n - val;
    return val;
}
