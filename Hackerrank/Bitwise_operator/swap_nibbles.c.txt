#include <stdio.h>
//Macro to swap nibbles
#define SWAP_NIBBLES(data) ((data & 0x0F)<<4 | (data & 0xF0)>>4)

#define MACRO_swap_nibbles(dat) ((dat >> 4) & 0x0F) | ((dat << 4) & 0xF0) 
int main()
{
    unsigned char value = 0x23; //value in hex
  
    printf("0x%x", MACRO_swap_nibbles(value)); //print after swapping
  
    return 0;
}
