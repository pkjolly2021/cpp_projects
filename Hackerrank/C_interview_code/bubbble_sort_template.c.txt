#include <iostream>
#include <string.h>
using namespace std;


template<typename T>
void bubbl_Sort(T& arr, int size)
{
    for(int i=0;i< size; i++)
    {
        for (int j=0; j<size;j++)
        {
            if(arr[j] > arr[j+1])
            {
                int tmp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = tmp;
            }
        }
    }
    
}



int main()
{
    //cout<<"Hello World";
    
        int arr[] = {10,12,8,5,21};
    //printf("Hello World");
     bubbl_Sort(arr, 5);
     for(int i=0;i<5;i++)
     {
         std::cout<< arr[i]<<',';
     }

     char str_arr[] = "debca";
     bubbl_Sort(str_arr, strlen(str_arr));
     for(int i=0;i<5;i++)
     {
         std::cout<< str_arr[i]<<',';
     }

    return 0;
}
