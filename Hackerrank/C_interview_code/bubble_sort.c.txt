#include <stdio.h>

void bubble_sort(int *arr, int size)
{
    for(int i=0;i< size; i++)
    {
        for (int j=0; j<size;j++)
        {
            if(arr[j] > arr[j+1])
            {
                int tmp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = tmp;
            }
        }
    }
}



int main()
{
    int arr[] = {10,12,8,5,21};
    //printf("Hello World");
     bubble_sort(arr, 5);
     for(int i=0;i<5;i++)
     {
         printf("%d", arr[i]);
         printf("%c", ',');
     }
    return 0;
}
