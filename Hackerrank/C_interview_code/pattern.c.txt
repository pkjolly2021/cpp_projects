
#include <stdio.h>

void print_patter(int max)
{
    for(int i=1; i<=max; i++)
    {
        for (int j=1;j<=i;j++)
        {
            printf("%d", j);
            printf("%c", ' ');
        }
        printf("\n");
    }
}

int main()
{
    //printf("Hello World");
    print_patter(10);
    return 0;
}
